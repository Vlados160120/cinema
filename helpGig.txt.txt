$ git init // команда инецеализирует локальный репозиторий (тоесть создает локальный git проект)полсе это команды нужно обезательно выполнить настройки пользователя
$ git config -- global user.name "Vlad"
$ git config -- global user.email "v16d012000@gmail.com"
$git config -- list//вывод всех глобальных настроек для проверок успешной настроек
$ git add .//добавит к git все файлы проекта если один конкретный файл то его надо указать место точки 
$ git status //команда проверяет состояние проекта или выполнение придедущих команд используется очень часто 
$ git help status// выводит html страницу со всеми командами
